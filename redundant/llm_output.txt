def convert_salary_estimate_to_float(df):
    """
    Convert the 'Salary Estimate' column to float type.
    
    This transformation is necessary because the 'Salary Estimate' column is currently of object type, 
    which may cause issues with numerical operations.
    """
    df_out = df.copy()
    change_log = []
    
    df_out['Salary Estimate'] = df_out['Salary Estimate'].str.replace('$', '').str.replace(',', '').astype(float)
